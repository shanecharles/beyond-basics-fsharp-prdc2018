open System.Text.RegularExpressions

let (|RegexMatch|_|) (pattern : string) v =
  let m = Regex.Match (v, pattern)
  if m.Success
  then [ for x in m.Groups do 
             yield x.Value ] 
       |> List.skip 1   
       |> Some
  else None
